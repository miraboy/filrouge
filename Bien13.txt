// Fonction pour supprimer les paragraphes avec les classes "test" ou "platypus"
function removeTestAndPlatypusParagraphs() {
    // Sélectionner et supprimer en une seule fonction
    $("p.test, p.platypus").remove();
}

// Configurer l'événement click sur le bouton
function setupButtonClick() {
    $("#removeButton").click(removeTestAndPlatypusParagraphs);
}

// Exécuter la configuration lorsque le document est prêt
$(document).ready(function() {
    setupButtonClick();
});
