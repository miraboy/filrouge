// Fonction fléchée ES6 pour styliser le premier élément avec la classe "platypus"
const stylePlatypus = () => {
    // Sélectionner le premier élément avec la classe "platypus"
    const platypusElement = document.querySelector('.platypus');
    
    // Vérifier si l'élément existe
    if (platypusElement) {
        // Appliquer un fond bleu
        platypusElement.style.backgroundColor = 'blue';
        platypusElement.style.color = 'white';
    }
};

// Exécuter la fonction lorsque le DOM est chargé
document.addEventListener('DOMContentLoaded', stylePlatypus);
