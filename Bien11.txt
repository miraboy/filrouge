// Fonction pour ajouter un élément
function addItem(inputElement) {
    // Récupérer la valeur de l'input
    const value = $(inputElement).val().trim();
    
    // Vérifier que la valeur n'est pas vide
    if (value) {
        // Créer un nouveau div avec la valeur
        const newDiv = $('<div>').addClass('added-item').text(value);
        
        // Ajouter le div après l'élément input
        $(inputElement).after(newDiv);
        
        // Vider l'input
        $(inputElement).val('');
    }
}
//____
// Configurer l'événement click sur le bouton
function setupButtonClick() {
    $("#addButton").click(function() {
        // Appeler addItem avec l'input en paramètre
        addItem($("#listItem"));
    });
}

// Exécuter la configuration lorsque le document est prêt
$(document).ready(function() {
    setupButtonClick();
});
