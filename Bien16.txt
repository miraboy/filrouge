// Fonction fléchée ES6 avec paramètre couleur hexadécimale
const setPlatypusColor = (hexColor = 'pink') => {
    // Sélectionner le premier élément avec la classe "platypus"
    const platypusElement = document.querySelector('.platypus');
    
    // Vérifier si l'élément existe
    if (platypusElement) {
        // Valider le format hexadécimal
        const hexRegex = /^#([A-Fa-f0-9]{6}|[A-Fa-f0-9]{3})$/;
        
        if (hexRegex.test(hexColor)) {
            // Appliquer la couleur si le format est valide
            platypusElement.style.backgroundColor = hexColor;
        } else {
            // Appliquer la couleur par défaut (pink) si le format est invalide
            platypusElement.style.backgroundColor = 'pink';
            console.warn('Couleur hexadécimale invalide, utilisation du rose par défaut');
        }
    }
};

// Fonction pour appliquer la couleur à partir de l'input
function applyColor() {
    const colorInput = document.getElementById('colorInput');
    const hexColor = colorInput.value.trim();
    setPlatypusColor(hexColor);
}

// Exécuter la fonction avec la valeur par défaut au chargement
document.addEventListener('DOMContentLoaded', () => {
    setPlatypusColor(); // Utilise la valeur par défaut (pink)
});
