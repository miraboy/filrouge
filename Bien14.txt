// Fonction pour configurer les événements sur les paragraphes
function setupParagraphEvents() {
    // Survol: ajouter la classe "blue"
    $("p").hover(
        function() {
            // Mouse enter - ajouter la classe blue
            $(this).addClass("blue");
        },
        function() {
            // Mouse leave - retirer la classe blue
            $(this).removeClass("blue");
        }
    );
    
    // Clic: basculer la classe "highlighted"
    $("p").click(function() {
        $(this).toggleClass("highlighted");
    });
}

// Exécuter la configuration lorsque le document est prêt
$(document).ready(function() {
    setupParagraphEvents();
});
