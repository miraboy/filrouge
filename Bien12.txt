// Fonction pour insérer du texte avant et après la première image
function insertTextAroundImage() {
    // Sélectionner la première image de la page
    const $firstImage = $("img:first");
    
    // Insérer le texte avant l'image
    $firstImage.before('<div class="inserted-text">Wow, I precede the image!</div>');
    
    // Insérer le texte après l'image
    $firstImage.after('<div class="inserted-text">Hey, I come in last</div>');
}

// Exécuter la fonction lorsque le document est prêt
$(document).ready(function() {
    insertTextAroundImage();
});
